
CREATE SCHEMA IF NOT EXISTS LEAGUEPONG;

USE LEAGUEPONG;

CREATE TABLE IF NOT EXISTS LIGAS(
    ID_LIGA BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    NOMBRE VARCHAR(255),
	PASSWORD VARCHAR(25),
    REGLAS VARCHAR(255),
     UBICACION VARCHAR(25),
     NUMERO_JUGADORES INT (50),
	CONSTRAINT NUMERO_JUGADORES CHECK (NUMERO_JUGADORES BETWEEN 1 AND 24)
	/*MINIMUM_LEVEL (INT) RANGO 1 A 4*/
);
