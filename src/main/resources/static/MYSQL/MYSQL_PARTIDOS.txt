CREATE SCHEMA IF NOT EXISTS LEAGUEPONG;

USE LEAGUEPONG;

CREATE TABLE IF NOT EXISTS PARTIDOS(
    ID_PARTIDO BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    ID_JUGADOR_1 BIGINT UNSIGNED,
    FOREIGN KEY (ID_JUGADOR_1) REFERENCES USUARIOS (ID_USUARIO),
    ID_JUGADOR_2 BIGINT UNSIGNED,
      FOREIGN KEY (ID_JUGADOR_2) REFERENCES USUARIOS (ID_USUARIO),
	ID_GANADOR BIGINT UNSIGNED,
      FOREIGN KEY (ID_GANADOR) REFERENCES USUARIOS (ID_USUARIO)
 
);
