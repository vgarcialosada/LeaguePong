CREATE SCHEMA IF NOT EXISTS LEAGUEPONG;

USE LEAGUEPONG;

CREATE TABLE IF NOT EXISTS USUARIOS_LIGAS(
    ID_USUARIO_LIGA BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    ID_USUARIO BIGINT UNSIGNED,
    FOREIGN KEY (ID_USUARIO) REFERENCES USUARIOS (ID_USUARIO),
    ID_LIGA BIGINT UNSIGNED,
        FOREIGN KEY (ID_LIGA) REFERENCES LIGAS (ID_LIGA), 
       IS_ADMIN BOOLEAN
 
);
